pragma solidity ^0.8.0;

contract Simple {
    uint256 public val1;
    uint256 public val2;

    constructor() {
        val2 = 3;
    }

    function set(uint256 _param) external {
        val1 = _param;
    }

- 目标地址未指定。
- 数据是初始化代码片段，其中包括合约的二进制代码。这是编译器生成的*.bin文件中的输出。

> - 初始化函数(`constructor`)不会存储在区块链上。
> - 交易的数据被视为合约部署的代码。
